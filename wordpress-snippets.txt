/**
 * Hook for <theme>/functions.php to hide uncategorized, tag ID 1, posts.
 **/
add_action('pre_get_posts', 'hide_uncategorized' );
function hide_uncategorized( $wp_query ) {
    set_query_var('category__not_in', '1');
}
